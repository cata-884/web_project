BASE = http://localhost/cat/public
TOKEN = (valoarea returnata de /api/auth/login → token)
ADMIN_TOKEN = (token al unui user cu role=admin)

Inlocuieste TOKEN, ADMIN_TOKEN, ID-urile si slug-urile cu valori reale la rulare.

# auth

# register
curl -s -X POST http://localhost/cat/public/api/auth/register \
  -H "Content-Type: application/json" \
  -d '{"username":"testuser","email":"test@example.com","password":"parola123"}' | jq .
# asteptat: {"success":true, "token":"...", "user":{...}}

# login
curl -s -X POST http://localhost/cat/public/api/auth/login \
  -H "Content-Type: application/json" \
  -d '{"email":"test@example.com","password":"parola123"}' | jq .
# asteptat: {"success":true, "token":"...", "user":{"id":...,"role":"user",...}}

# me (profil curent)
curl -s http://localhost/cat/public/api/auth/me \
  -H "Authorization: Bearer TOKEN" | jq .
# asteptat: {"id":1,"username":"testuser","email":"test@example.com","role":"user",...}

# update profil
curl -s -X PATCH http://localhost/cat/public/api/users/me \
  -H "Authorization: Bearer TOKEN" \
  -H "Content-Type: application/json" \
  -d '{"full_name":"Prenume Nume","email":"test@example.com"}' | jq .
# asteptat: {"success":true,"user":{...}}

# logout
curl -s -X POST http://localhost/cat/public/api/auth/logout \
  -H "Authorization: Bearer TOKEN" | jq .
# asteptat: {"success":true}

    "description": "Camping la poalele cheilor, aproape de trasee",

    "longitude": 23.6897,

    "capacity": 50
